<script>
const startHour = 8;
const endHour = 22;
const target = 34;

const tableBody = document.getElementById("tableBody");
let labels = [];
let avgData = [];

for (let hour = startHour; hour <= endHour; hour++) {
    const timeLabel = hour + ":00";
    labels.push(timeLabel);
    avgData.push(0);

    const row = document.createElement("tr");
    row.innerHTML = `
        <td>${timeLabel}</td>
        <td><input type="number" min="0" max="100" oninput="calculateRow(this)"></td>
        <td><input type="number" min="0" max="100" oninput="calculateRow(this)"></td>
        <td><input type="number" min="0" max="100" oninput="calculateRow(this)"></td>
        <td class="avg-cell">0.00</td>
    `;
    tableBody.appendChild(row);
}

function calculateRow(input){
    const row = input.closest("tr");
    const inputs = row.querySelectorAll("input");
    const avgCell = row.querySelector(".avg-cell");

    let values = [];
    inputs.forEach(i=>{
        let val=parseFloat(i.value);
        if(!isNaN(val)) values.push(val);
    });

    let avg = values.length ? values.reduce((a,b)=>a+b,0)/values.length : 0;
    avgCell.innerText = avg.toFixed(2);

    updateChart();
    saveDataForDate(); // 🔥 AUTO SAVE
}


function updateChart(){
    const rows = document.querySelectorAll("#tableBody tr");
    avgData = [];

    rows.forEach(row=>{
        avgData.push(parseFloat(row.querySelector(".avg-cell").innerText));
    });

    chart.data.datasets[0].data = avgData;
    chart.update();
}

const ctx = document.getElementById('trendChart').getContext('2d');

const chart = new Chart(ctx,{
    type:'line',
    data:{
        labels:labels,
        datasets:[
            {
                label:'Average Performance',
                data:avgData,
                borderColor:'#16a34a',
                backgroundColor:'rgba(22,163,74,0.15)',
                fill:true,
                tension:0.4,
                pointRadius:5
            },
            {
                label:'Target (34%)',
                data:Array(labels.length).fill(target),
                borderColor:'#dc2626',
                borderDash:[6,6],

                pointRadius:0
            }
        ]
    },
    options:{
        responsive:true,
        maintainAspectRatio:false,
        scales:{
            y:{
                min:0,
                max:100,
                ticks:{
                    callback:(v)=>v+"%"
                }
            }
        },
        plugins:{
            legend:{
                labels:{
                    font:{size:13}
                }
            }
        }
    }
});

function extractCSV(){

    const selectedDate = document.getElementById("monitoringDate").value;

    if(!selectedDate){
        alert("Please select a date before extracting data.");
        return;
    }

    let csv=[];
    csv.push(["SPRH Monitoring VISMIN"].join(","));
    csv.push(["Date:", selectedDate].join(","));
    csv.push([""].join(","));
    csv.push(["Time","PNC %","MAA %","CGY %","Avg %"].join(","));

    const rows=document.querySelectorAll("#tableBody tr");

    rows.forEach(row=>{
        const time=row.children[0].innerText;
        const inputs=row.querySelectorAll("input");
        const avg=row.querySelector(".avg-cell").innerText;

        let rowData=[
            time,
            inputs[0].value||0,
            inputs[1].value||0,
            inputs[2].value||0,
            avg
        ];
        csv.push(rowData.join(","));
    });

    function saveToGoogleSheet(){

    const selectedDate = dateInput.value;
    if(!selectedDate){
        alert("Please select a date.");
        return;
    }

    const rows = document.querySelectorAll("#tableBody tr");

    let dataToSend = {
        date: selectedDate,
        rows: []
    };

    rows.forEach(row=>{
        const time = row.children[0].innerText;
        const inputs = row.querySelectorAll("input");
        const avg = row.querySelector(".avg-cell").innerText;

        dataToSend.rows.push({
            time: time,
            pnc: inputs[0].value || "",
            maa: inputs[1].value || "",
            cgy: inputs[2].value || "",
            avg: avg
        });
    });

    fetch("https://script.google.com/macros/s/AKfycbwI0oGqg5DEbNA_lng3ceb3GkIIzEUOS_9utZOrWuxLzDgyyeA7PYPfb9XOi7ddCR8HFA/exec?date=2026-02-13
", {
        method: "POST",
        headers: {

        },
        body: JSON.stringify(dataToSend)
    })
    .then(res => res.json())

        alert("Saved successfully!");
    })

        console.error(err);
        alert("Error saving data.");
    });
}


    const csvContent="data:text/csv;charset=utf-8,"+csv.join("\n");
    const encodedUri=encodeURI(csvContent);
    const link=document.createElement("a");

    link.setAttribute("href",encodedUri);

const dateInput = document.getElementById("monitoringDate");

// Auto set today's date
dateInput.valueAsDate = new Date();

loadDataFromGoogleSheet();

function saveDataForDate(){

    const selectedDate = dateInput.value;
    if(!selectedDate) return;

    const rows = document.querySelectorAll("#tableBody tr");
    let data = [];

    rows.forEach(row=>{
        const inputs = row.querySelectorAll("input");
        data.push({
            pnc: inputs[0].value || "",
            maa: inputs[1].value || "",
            cgy: inputs[2].value || ""
        });
    });

    localStorage.setItem("SPRH_"+selectedDate, JSON.stringify(data));
}


function loadDataForDate(){

    const selectedDate = dateInput.value;
    if(!selectedDate) return;

    const saved = localStorage.getItem("SPRH_"+selectedDate);

    const rows = document.querySelectorAll("#tableBody tr");

    if(saved){
        const data = JSON.parse(saved);

        rows.forEach((row,index)=>{
            const inputs = row.querySelectorAll("input");

            inputs[0].value = data[index]?.pnc || "";
            inputs[1].value = data[index]?.maa || "";
            inputs[2].value = data[index]?.cgy || "";

            calculateRow(inputs[0]);
        });

    } else {
        // Clear table if no saved data
        rows.forEach(row=>{
            const inputs = row.querySelectorAll("input");
            inputs.forEach(input=> input.value="");
            row.querySelector(".avg-cell").innerText = "0.00";
        });

        updateChart();
    }
}


function loadDataFromGoogleSheet(){

    const selectedDate = dateInput.value;
    if(!selectedDate) return;

    fetch("https://script.google.com/macros/s/AKfycbwI0oGqg5DEbNA_lng3ceb3GkIIzEUOS_9utZOrWuxLzDgyyeA7PYPfb9XOi7ddCR8HFA/exec?date=2026-02-13
"+selectedDate)
    .then(res => res.json())
    .then(data => {

        const rows = document.querySelectorAll("#tableBody tr");

        rows.forEach(row=>{
            const inputs = row.querySelectorAll("input");
            inputs.forEach(input => input.value="");
            row.querySelector(".avg-cell").innerText="0.00";
        });

        data.forEach((item,index)=>{
            const row = rows[index];
            if(!row) return;

            const inputs = row.querySelectorAll("input");

            inputs[0].value = item.pnc;
            inputs[1].value = item.maa;
            inputs[2].value = item.cgy;

            calculateRow(inputs[0]);
        });

    })
    .catch(err=>{
        console.error("Load error:",err);
    });
}
